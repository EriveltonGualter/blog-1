Cloud router with TP-Link TL-MR3020

Posted on July 20, 2016 in misc • 1 min read

make image PROFILE=TLMR3020 PACKAGES="kmod-usb-storage block-mount kmod-fs-ext4 kmod-scsi-core hd-idle portmap nfs-kernel-server"

telnet 192.168.1.1

passwd

ssh root@192.168.1.1

mkdir /storage mount /dev/sda /storage

vi /etc/config/fstab

config 'global' option anon_swap '0' option anon_mount '0' option auto_swap '1' option auto_mount '1' option delay_root '5' option check_fs '0'

config 'mount' option target '/mnt/sda' option uuid '1eff386b-1c5c-44d8-97e9-e3f2eab45a65' option enabled '1'

Change enabled to 1.

vi /etc/exports

/storage *(rw,fsid=root,no_subtree_check,root_squash,async,insecure)

Reload config

exportfs -ar

Open some ports

iptables -A INPUT -p tcp --dport 111 -m state --state NEW -j ACCEPT iptables -A INPUT -p udp --dport 111 -m state --state NEW -j ACCEPT iptables -A INPUT -p tcp --dport 32777:32780 -m state --state NEW -j ACCEPT iptables -A INPUT -p udp --dport 32777:32780 -m state --state NEW -j ACCEPT

Start NFS

root@OpenWrt:~# /etc/init.d/portmap start root@OpenWrt:~# /etc/init.d/nfsd start root@OpenWrt:~# /etc/init.d/portmap enable root@OpenWrt:~# /etc/init.d/nfsd enable

PC

Check if NFS works

sudo showmount -e 192.168.1.1

Mount

mkdir /home/alexandre/storage sudo mount -rw -v -t nfs -o vers=3,nfsvers=3 192.168.1.1:/storage /home/alexandre/storage
